00037 QImage GetQImage(const TImage* image, const itk::ImageRegion<2>& region, const DisplayStyle& style)
00038 {
00039   // Expects an itk::VectorImage
00040   switch(style.Style)
00041   {
00042     case DisplayStyle::COLOR:
00043       return GetQImageColor<TImage>(image, region);
00044       break;
00045     case DisplayStyle::MAGNITUDE:
00046       return GetQImageMagnitude<TImage>(image, region);
00047       break;
00048     case DisplayStyle::CHANNEL:
00049       {
00050       typedef itk::Image<typename TImage::InternalPixelType, 2> ScalarImageType;
00051       typename ScalarImageType::Pointer channelImage = ScalarImageType::New();
00052       ITKHelpers::ExtractChannel<typename TImage::InternalPixelType>(image, style.Channel, channelImage);
00053       return GetQImageScalar<ScalarImageType>(channelImage, region);
00054       break;
00055       }
00056     default:
00057       std::cerr << "No valid style specified." << std::endl;
00058       return QImage();
00059   }
00060   return QImage();

00063 template <typename TImage>
00064 QImage GetQImageColor(const TImage* image, const itk::ImageRegion<2>& region)
00065 {
00066   // Get a color QImage from an ITK image.
00067   QImage qimage(region.GetSize()[0], region.GetSize()[1], QImage::Format_RGB888);
00068 
00069   typedef itk::RegionOfInterestImageFilter< TImage, TImage > RegionOfInterestImageFilterType;
00070   typename RegionOfInterestImageFilterType::Pointer regionOfInterestImageFilter = RegionOfInterestImageFilterType::New();
00071   regionOfInterestImageFilter->SetRegionOfInterest(region);
00072   regionOfInterestImageFilter->SetInput(image);
00073   regionOfInterestImageFilter->Update();
00074 
00075   itk::ImageRegionIterator<TImage> imageIterator(regionOfInterestImageFilter->GetOutput(), regionOfInterestImageFilter->GetOutput()->GetLargestPossibleRegion());
00076 
00077   while(!imageIterator.IsAtEnd())
00078     {
00079     typename TImage::PixelType pixel = imageIterator.Get();
00080 
00081     itk::Index<2> index = imageIterator.GetIndex();
00082     int r = static_cast<int>(pixel[0]);
00083     int g = static_cast<int>(pixel[1]);
00084     int b = static_cast<int>(pixel[2]);
00085     
00086     if(Helpers::IsValidRGB(r,g,b))
00087       {
00088       QColor pixelColor(r,g,b);
00089       qimage.setPixel(index[0], index[1], pixelColor.rgb());
00090       }
00091     else
00092       {
00093       std::cerr << "Can't set r,g,b to " << r << " " << g << " " << b << std::endl;
00094       QColor pixelColor(0,0,0);
00095       qimage.setPixel(index[0], index[1], pixelColor.rgb());
00096       }
00097 
00098     ++imageIterator;
00099     }
00100 
00101   QColor highlightColor(255, 0, 255);
00102   qimage.setPixel(region.GetSize()[0]/2, region.GetSize()[1]/2, highlightColor.rgb());
00103 
00104   //return qimage; // The actual image region
00105   return qimage.mirrored(false, true); // The flipped image region
00106 }
