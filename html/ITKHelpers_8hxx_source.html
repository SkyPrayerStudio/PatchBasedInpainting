00222 
00223   // 'Mask' must be the same size as 'image'

00236   itk::ImageRegionConstIterator<Mask> maskIterator(mask,region);
00237   itk::ImageRegionIterator<TImage> imageIterator(image, region);
00238 
00239   while(!patchIterator.IsAtEnd())
00240     {
00241     if(mask->IsHole(maskIterator.GetIndex())) // we are in the target region
