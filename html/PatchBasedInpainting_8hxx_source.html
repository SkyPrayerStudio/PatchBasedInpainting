00050 template <typename TImage>
00051 PatchBasedInpainting<TImage>::PatchBasedInpainting(const TImage* const image, const Mask* const mask)
00052 {
00053   EnterFunction("PatchBasedInpainting()");
00054 
00055   this->PatchRadius.Fill(3);
00056 
00057   // We don't want to modify the input images, so we copy them.
00058   this->MaskImage = Mask::New();
00059   this->MaskImage->DeepCopyFrom(mask);
00060 
00061   this->CurrentInpaintedImage = TImage::New();
00062   ITKHelpers::DeepCopy<TImage>(image, this->CurrentInpaintedImage);
00063 
00064   ColorImageInsideHole();
00065 
00066   this->FullImageRegion = image->GetLargestPossibleRegion();
00067   if(this->MaskImage->GetLargestPossibleRegion() != this->FullImageRegion)
00068     {
00069     std::stringstream ss;
00070     ss << "Mask and image size must match! Mask is " << this->MaskImage->GetLargestPossibleRegion().GetSize()
00071               << " while image is " << this->FullImageRegion << std::endl;
00072     throw std::runtime_error(ss.str());
00073     }
00074 
00075   // We definitely want to update the image and the mask. We can add more images to the list to update later if necessary.
00076   ImagesToUpdate.push_back(this->CurrentInpaintedImage);
00077   ImagesToUpdate.push_back(this->MaskImage); // We MUST update the mask LAST, because it is used to know where to update everything else!
00078 
00079   // Set the image to use for pixel to pixel comparisons.
00080   this->CompareImage = this->CurrentInpaintedImage;
00081 
00082   // Set defaults
00083   this->NumberOfCompletedIterations = 0;
00084 
00085   this->PriorityFunction = NULL; // Can't initialize this here, must wait until the image and mask are opened
00086 }
